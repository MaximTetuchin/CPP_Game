80void Player::update(float time) {
81 // Update speed boost timer
82 if(isSpeedBoosted) {
83 speedBoostTime -= time;
84 if(speedBoostTime <= 0) {
85 speedMultiplier = 1.0f;
86 isSpeedBoosted = false;
87 }
88 }
89
90 // Update invincibility timer
91 if(isInvincible) {
92 invincibilityTime -= time;
93 if(invincibilityTime <= 0) {
94 isInvincible = false;
95 }
96 }
97
98 // Horizontal movement and collision
99 rect.left += dx * time;
100 Collision(0);
101
102 // Vertical movement with gravity
103 if (!onGround) dy += 0.0005 * time;
104 rect.top += dy * time;
105 onGround = false;
106 Collision(1);
107
108 // Animation frames
109 currentFrame += 0.005 * time;
110 idleFrame = currentFrame / 1.5;
111
112 if (currentFrame > 6) currentFrame -= 6;
113 if (idleFrame > 3) idleFrame -= 3;
114
115 // Set sprite texture based on movement direction
116 if (dx > 0) {
117 sprite.setTextureRect(sf::IntRect(40 * int(currentFrame), 244, 40, 50));
118 lastAction = true;
119 }
120 else if (dx < 0) {
121 sprite.setTextureRect(sf::IntRect(40 * int(currentFrame) + 40, 244, -40, 50));
122 lastAction = false;
123 }
124 else {
125 // Idle animation
126 if (lastAction) {
127 sprite.setTextureRect(sf::IntRect(43 * int(idleFrame) + 4, 188, 40, 50));
128 } else {
129 sprite.setTextureRect(sf::IntRect(43 * int(idleFrame) + 4 + 40, 188, -40, 50));
130 }
131 }
132
133 sprite.setPosition(rect.left, rect.top);
134
135 // Update weapon
136 if (currentWeapon) {
137 currentWeapon->update(time);
138 }
139 dx = 0;
140}

149void Player::Collision(int dir) {
150 for (int i = rect.top / 32; i < (rect.top + rect.height) / 32; i++) {
151 if (i < 0 || i >= H) continue;
152
153 for (int j = rect.left / 32; j < (rect.left + rect.width) / 32; j++) {
154 if (j < 0 || j >= W) continue;
155
156 if (TileMap[i][j] == 'B' || TileMap[i][j] == 'G' || TileMap[i][j] == 'W' || TileMap[i][j] == 'Z') {
157 if (TileMap[i][j] == 'Z') {
158 health = 100;
159 respawn();
160 }
161 if ((dx > 0) && (dir == 0)) rect.left = j * 32 - rect.width;
162 if ((dx < 0) && (dir == 0)) rect.left = j * 32 + 32;
163 if ((dy > 0) && (dir == 1)) {
164 rect.top = i * 32 - rect.height;
165 dy = 0;
166 onGround = true;
167 }
168 if ((dy < 0) && (dir == 1)) {
169 rect.top = i * 32 + 32;
170 dy = 0;
171 }
172 }
173 if (TileMap[i][j] == '0') TileMap[i][j] = ' ';
174 }
175 }
